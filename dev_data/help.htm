Вывод данных в JS, HTML

Можно загружать данные с config.live.json, Cookies, lang.*.json


jsonResponse.ssid;
// В данном случае мы получим имя WiFi сети к которой мы подключены
// Пример:
alert(jsonResponse.ssid);

// Если вы хотите вывести эти данные в HTML это необходимо сделать следующим способом:
{{ssid}}
// Пример:
<h3>{{ssid}}</h3>

Ajax GET запросы


send_request(this,'/test');
// В данном случае мы отправим GET на /test
// Пример:
<input class="btn btn-block btn-success" value="button" onclick="send_request(this,'/test');" type="button">

Работы с input полями


val('ID_name');

// Пример:
<input class="form-control" value="" id="param">
<input class="btn btn-block btn-success" value="button" onclick="send_request(this,'/test='+val('param'));" type="button">
// Нажав на кнопку мы отпрваим ajax get запрос на /test=значение_с_поля_param
